for (i in 1:v) {
  trainFold = as.integer(folds[, -i])
  testFold = folds[, i]
  
  for (j in 1:length(cps)) {
    tree = rpart(`winner` ~ .,
            data = eResultTrain[trainFold, -c(2,3)], 
            method = "class",
            control = rpart.control(cp = cps[j]))
    preds[ testFold,j ] = 
      predict(tree, 
              newdata = eResultTest[testFold,-c(1,2,3)],
              type = "class")
  }
}
